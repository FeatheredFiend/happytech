SELECT feedback_response.id, template.name AS template, feedback_type.name AS feedback, applicant.name AS applicant, statement.statement
FROM applicant
    LEFT JOIN feedback_response ON feedback_response.applicant_id = applicant.id
    LEFT JOIN template ON feedback_response.template_id = template.id
    LEFT JOIN template_header ON template.header_id = template_header.id
    LEFT JOIN feedback_type ON template_header.feedbacktype_id = feedback_type.id
    LEFT JOIN template_statement ON template_statement.template_id = template.id
    LEFT JOIN statement ON template_statement.statement_id = statement.id
WHERE template.id = 3

SELECT template.name, user.name, feedback_type.name, statement.statement, applicant.name
FROM applicant
    LEFT JOIN feedback_response ON feedback_response.applicant_id = applicant.id
    LEFT JOIN template ON feedback_response.template_id = template.id
    LEFT JOIN template_statement ON template_statement.template_id = template.id
    LEFT JOIN statement ON template_statement.statement_id = statement.id
    LEFT JOIN template_header ON template.header_id = template_header.id
    LEFT JOIN feedback_type ON template_header.feedbacktype_id = feedback_type.id
    LEFT JOIN user ON template_header.user_id = user.id
    WHERE applicant.id = 1
